2 . Enable WSL (Window setup for linux) in app & feature settings
3 . Install linux app in MS store
4 . sudo apt-get install software-properties-common
5 . sudo apt-add-repository universe
6 . sudo apt-get update
7 . sudo apt-get install python-setuptools
8 . sudo apt install python3-pip
9 . sudo apt upgrade
10 . sudo apt-get install libmysqlclient-dev
11 . sudo apt-get install libssl-dev
12 . sudo apt-get install libkrb5-dev
13 . pip3 install apache-airflow
14 . sudo nano /etc/wsl.conf
15 . ADD in -> [automount]
		root = / 
		options = "metadata"
16 . nano ~/.bashrc - >   export AIRFLOW_HOME = c/airflow
	=> source ~/.bashrc
17 . sudo apt update  - >   cd airflow
18 . ls
19 . nano airflow.cfg  -> run if present in ls  -> Change dag path where you want.
20 . airflow db init
21 . airflow users create --username admin --firstname admin --lastname admin --role Admin --email admin 
	  "airflow delete -u <username>"
22 . airflow webserver -p 8080
23 . login with user name and password.
24 . airflow scheduler in another tap.
25 . pip install mysql-client  
     ->   pip install mysql-connector-python
     ->   pip install mysql-connector
     ->   pip install mysql
     ->   pip install apache-airflow-providers-mysql

26 . In mysql work bench

# drop database air;
# drop user if exists 'air'@'%' ;
CREATE DATABASE air;
CREATE USER 'air'@'%' IDENTIFIED BY 'password';
GRANT USAGE,EXECUTE ON *.* TO 'air'@'%';
GRANT ALL PRIVILEGES ON air.* TO 'air'@'%';
FLUSH PRIVILEGES;

27 . In airflow.cfg - > sql_alchemy_conn = mysql://air:password@<host>:<port>/air
28 . cd -e -> airflow db init
29 . airflow webserver , airflow scheduler.
30 . sudo chmod 777 dags
