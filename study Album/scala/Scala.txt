Polymorphism:
	- one interface, multiple implementation. one of the job doing in a different way.

	--------------
	class Animal {

	class Dog extends Animal {
	  override def makeSound(): Unit = println("Woof! Woof!")
	}

	class Cat extends Animal {
	  override def makeSound(): Unit = println("Meow")
	}

	  animal.makeSound()
	}

	animalSound(cat) // Output: Meow

Encapsulation:
	- Data hiding and controlled access. (public, protected, private)
	- providing access to the data only through defined methods
	- eg:
	class Person(private var name: String, private var age: Int) {
		def setName(newName: String): Unit = {
		name = newName
		}

		def setAge(newAge: Int): Unit = {
		if (newAge >= 0) {
		  age = newAge
		} else {
		  println("Invalid age. Age cannot be negative.")
		}
		}

		def getInfo: String = {
		s"Name: $name, Age: $age"
		}
		}

		object Main {
		def main(args: Array[String]): Unit = {
		val person = new Person("Alice", 30)
		// person.name   // This would give a compilation error
		person.setName("Bob")
		person.setAge(-5)  // This will print "Invalid age. Age cannot be negative."
		println(person.getInfo) // Output: Name: Bob, Age: 30
		}
		}
